tipos de dados e leitura da linguagem C#:
tipos de dados básicos são semelhantes ao C. char, float, int, double..

Operadores:
simbolos pra relizar operações lógicas e aritméticas sobre operandos.

| Prioridade | Operador | Operação          |
| ---------- | -------- | ----------------- |
| 1          | -        | inversão de sinal |
| 2          | *        | multiplicação     |
| 2          | /        | divisão           |
| 2          | %        | resto da divisão  |
| 3          | +        | adição            |
| 3          | -        | subtração         |
para casos de prioridade equivalentes vai pela ordem de acontecimento (esquerda→direita) (igual para qualquer equação matemática normal)
O uso dos parênteses define a ordem prioritária da equação.

```
namespace SegundoProjetoPoggers
{
    internal class Program
    {
        static void Main(string[] args)
        {
            double nota1, nota2, nota3, media;
            Console.WriteLine("Atribuindo 3 notas para o aluno!");
            nota1 = 7.5;
            Console.WriteLine("Nota 1 = " + nota1);
            nota2 = 9.6;
            Console.WriteLine("Nota 2 = " + nota2);
            nota3 = 3.2;
            Console.WriteLine("Nota 3 = " + nota3);
            media = (nota1 + nota2 + nota3) / 3;
            Console.WriteLine("A média do aluno é: " +media.ToString("N2"));

            Console.WriteLine("===================================");

            int i;
            i = 0;
            Console.WriteLine("i = "+i);
            i = i + 1;
            Console.WriteLine("i = "+i);
            i++; //igual a i = i + 1;, mas só funciona de 1 em 1
            Console.WriteLine("i = "+i);
            i += 1; //igual a i++ e igual a i = i + 1;
            Console.WriteLine("i = "+i);
            i += 2; //quero adicionar de 2 em 2 para mais
            Console.WriteLine("i = "+i);
            i = i + 3; 
            Console.WriteLine("i = "+i);

            Console.WriteLine("===================================");

            Console.WriteLine("Primeiro projeto C#");
            Console.WriteLine("Vamos iniciar, manipulando strings");
            string frase = "Frase da declaração de variavel com string";
            String frase2 = "Frase da declaração de variavel com String";
            //concatenando string
            Console.WriteLine("Concatenando strings");
            Console.WriteLine("Frase 1 = "+frase);
            Console.WriteLine("Frase 2 = "+frase2);
            //interporlação de string
            Console.WriteLine("Interpolando de strings");
            Console.WriteLine($"Frase 1 = {frase}");
            Console.WriteLine($"Frase 2 = {frase2}");
            //Para comentar bloco, seleciona o bloco e pressiona ctrl + k e depois ctrl + c
            //Para descomentar bloco, seleciona o bloco e pressiona ctrl +k e depois ctrl + u
        }
    }
}


```

Comando de entrada: "Console.ReadLine()"
